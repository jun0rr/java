1     MailServer server = new MailServer()
   2         .setName("gmail_imap")
   3         .setServer("imap.gmail.com")
   4         .setProtocol(MailServer.PROTOCOL_IMAPS)
   5         .setPort(MailServer.DEFAULT_IMAP_PORT)
   6         .setAuthentication("juno.rr@gmail.com", "passwd=$0988");
   7     
   8     SMail mail = new SMail(server);
   9     MailBox box = null;
  10     try {
  11       box = mail.receive("INBOX");
  12     } catch(SMailException ex) {
  13       //Capture e trate o erro adequadamente.
  14       ex.printStackTrace();
  15     }
  16     
  17     //Recuperando todas as mensagens de "INBOX"
  18     Messages all = box.getMessages();
  19     
  20     //Podemos obter uma lista ou um array com as mensagens
  21     List<Message> list = all.getList();
  22     Message[] array = all.toArray();
  23     
  24     //Recuperando somente as mensagens novas
  25     Messages news = box.getNewMessages();
  26     
  27     //Pesquisando mensagens por assunto
  28     Messages ms = all.filter(new SubjectFilter("hello"));
  29     
  30     //Pesquisando mensagens por remetente
  31     ms = all.filter(new AddressFilter(
  32         AddressFilter.ADDRESS.FROM, "Duke"));
  33     
  34     //Pesquisando mensagens por data de envio
  35     ms = all.filter(new DateMessageFilter(
  36         DateMessageFilter.COMPARE.BEFORE, new Date()));
  37     
  38     //Pesquisando mensagens com anexos
  39     ms = all.filter(new HasAttachmentFilter());
  40     
  41     //Combinando filtros de pesquisa
  42     ms = all.filter(new OldMessageFilter())
  43         .filter(new HasAttachmentFilter())
  44         .filter(new ContentFilter("fotos das férias"));
  45     
  46     //Salvando anexos das mensagens
  47     Message fotos = ms.first();
  48     try {
  49       fotos.getAttachments().first().saveTo(
  50           new File("/home/juno/fotos/foto1.jpg"));
  51     } catch(IOException ex) {
  52       //Capture e trate o erro adequadamente.
  53       ex.printStackTrace();
  54     }
  55     
  56     //Marcando mensagens novas como lidas
  57     box.read(news);
  58     
  59     //Apagando mensagens
  60     box.delete(ms);
  61     
  62     //Configurando MailBox para notificar a chegada de novos emails
  63     MailBoxListener listener = new MailBoxListener() {
  64       public void messageReceived(MailBoxEvent e) {
  65         System.out.println(" [NEW MESSAGE RECEIVED!]");
  66       }
  67       public void reloadingMailBox(MailBoxEvent e) {}
  68     };
  69     
  70     box.add(listener);
  71     //15 minutos
  72     box.setAutoReload(15 * 60 * 1000);
  73     //dado o longo intervalo, é interessante 
  74     //que a conexão seja fechada após a verificação.
  75     //Não seria o caso se o tempo fosse de 15 segundos, por exmplo.
  76     box.setAutoCloseOnReload(true);
